import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from scipy.optimize import curve_fit

df = pd.read_csv('3.txt', sep='\\s+', header=0)
print("")
print(df.head())

x = df['x']
y = df['y']

# 
def neg_exp(x, b0, b1):
    return b0 * (1 - np.exp(-b1 * x))

initial_guess = [1.0, 0.01]  # 
params, cov = curve_fit(neg_exp, x, y, p0=initial_guess)

b0, b1 = params
print(f"\nb0 = {b0:.4f}, b1 = {b1:.4f}")

y_pred = neg_exp(x, *params)
ss_res = np.sum((y - y_pred)**2)
ss_tot = np.sum((y - np.mean(y))**2)
r2 = 1 - (ss_res / ss_tot)
print(f" R = {r2:.4f}")

plt.figure(figsize=(8,5))
plt.scatter(x, y, label='', color='royalblue')
plt.plot(x, y_pred, color='red', label=f': y={b0:.3f}(1-e^(-{b1:.3f}x))')
plt.xlabel("X")
plt.ylabel("Y")
plt.title("")
plt.legend()
plt.grid(alpha=0.3)
plt.show()
